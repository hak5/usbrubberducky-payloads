EXTENSION PASSFILE
    REM VERSION 1.0
    REM AUTHOR: Mr.Ender
    
    REM_BLOCK DOCUMENTATION
        USAGE:
            Call PASSFILE() at the beginning of your payload
            to check if a file named "PASSWORD" exists on the target Mac.
            The file can be hidden or visible.
            
        TARGETS:
            macOS systems with DuckyScript 3.0+
            
        CONFIGURATION:
            Change the file path by modifying the #PassFile_PATH define below
            Exsample paths:
                - /Users/USERNAME/PASSWORD
                - /Users/USERNAME/Documents/PASSWORD
                - /Users/USERNAME/Downloads/PASSWORD
                - /tmp/PASSWORD
    END_REM

    DEFINE #PassFile_PATH /Users/USERNAME/PASSWORD

    FUNCTION PASSFILE()
        CMD SPACE
        DELAY 1000
        STRINGLN terminal
        DELAY 1500
        STRINGLN [ -f '#PassFile_PATH' ] && (defaults write /Library/Preferences/com.apple.bluetooth.local.device -dict APDUDucky -bool true; killall Finder)
        DELAY 500
    END_FUNCTION
END_EXTENSION
